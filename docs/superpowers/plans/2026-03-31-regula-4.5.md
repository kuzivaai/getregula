```python
_MAX_PATTERN_LENGTH = 500  # Reject overly complex patterns

def _compile_custom_pattern(pattern: str) -> "re.Pattern":
    """Compile a custom rule pattern with basic ReDoS protection."""
    if len(pattern) > _MAX_PATTERN_LENGTH:
```

In `scripts/ast_analysis.py`, add before the `_FlowTracer` class (around line 385):

```python
class _CallGraphBuilder(ast.NodeVisitor):
    """Build a per-file function call graph and track return origins."""

    def __init__(self, ai_call_lines: Set[int]) -> None:
        self._ai_call_lines = ai_call_lines
        # function_name -> set of function_names it calls
        self.calls: Dict[str, Set[str]] = {}
        # function_name -> True if the function returns AI data
        self.returns_ai: Dict[str, bool] = {}
        # All function names defined in this file
        self.defined_functions: Set[str] = set()
        self._current_function: Optional[str] = None

    def visit_FunctionDef(self, node: ast.FunctionDef) -> None:
        self._visit_func(node)

    def visit_AsyncFunctionDef(self, node: ast.AsyncFunctionDef) -> None:
        self._visit_func(node)

    def _visit_func(self, node) -> None:
        prev = self._current_function
        self._current_function = node.name
        self.defined_functions.add(node.name)
        self.calls.setdefault(node.name, set())
        self.returns_ai.setdefault(node.name, False)

        # Check if any AI call line is inside this function
        for child in ast.walk(node):
            if hasattr(child, 'lineno') and child.lineno in self._ai_call_lines:
                self.returns_ai[node.name] = True
                break

        # Check return statements — if they return a call to another function
        # that returns AI data, this function also returns AI data
        # (resolved in a second pass)

        self.generic_visit(node)
        self._current_function = prev

    def visit_Call(self, node: ast.Call) -> None:
        if self._current_function:
            callee = _unparse_safe(node.func).split("(")[0].split(".")[-1]
            self.calls[self._current_function].add(callee)
        self.generic_visit(node)

    def resolve_transitive_ai(self) -> None:
        """Propagate AI returns transitively through the call graph."""
        changed = True
        while changed:
            changed = False
            for func, callees in self.calls.items():
                if self.returns_ai.get(func):
                    continue
                for callee in callees:
                    if self.returns_ai.get(callee):
                        self.returns_ai[func] = True
                        changed = True
                        break
```

In `_value_origin`, add a check for calls to AI-returning functions:

```python
def _value_origin(self, node: ast.AST) -> Optional[int]:
    if isinstance(node, ast.Name):
        return self._var_origins.get(node.id)
    if isinstance(node, ast.Call):
        line = getattr(node, "lineno", None)
        if line in self._ai_lines:
            return line
        # Cross-function: call to a function that returns AI data
        callee = _unparse_safe(node.func).split("(")[0].split(".")[-1]
        if callee in self._ai_returning_functions:
            # Create a synthetic AI origin for this call
            if line and line not in self._ai_lines:
                self._ai_lines[line] = {
                    "source": f"{callee}() [cross-function]",
                    "source_line": line,
                }
                self._destinations[line] = []
            return line
    if isinstance(node, ast.Attribute):
        return self._value_origin(node.value)
    if isinstance(node, ast.Subscript):
        return self._value_origin(node.value)
    return None
```

```python
def trace_ai_data_flow(content: str) -> List[Dict]:
    try:
        tree = ast.parse(content, mode="exec")
    except SyntaxError:
        return []

    ai_calls = _find_ai_calls(tree)
    ai_call_lines = {c["source_line"] for c in ai_calls}

    # Build call graph to find functions that transitively return AI data
    cg = _CallGraphBuilder(ai_call_lines)
    cg.visit(tree)
    cg.resolve_transitive_ai()
    ai_funcs = {f for f, returns in cg.returns_ai.items() if returns}

    tracer = _FlowTracer(ai_calls, ai_returning_functions=ai_funcs)
    return tracer.analyse(tree)
```

```python
"""Content-hash scan caching for Regula.

Caches scan results keyed by SHA-256 of file content.
Unchanged files skip re-scanning on subsequent runs.
"""
import hashlib
import json
from pathlib import Path
from typing import Optional


class ScanCache:
    def __init__(self, cache_dir: Optional[Path] = None):
        self._cache_dir = cache_dir or Path.home() / ".regula" / "cache"
        self._cache_dir.mkdir(parents=True, exist_ok=True)
        self._memory: dict = {}  # In-memory for current session
        self._load()

    def _cache_file(self) -> Path:
        return self._cache_dir / "scan_cache.json"

    def _load(self) -> None:
        cf = self._cache_file()
        if cf.exists():
            try:
                self._memory = json.loads(cf.read_text(encoding="utf-8"))
            except (json.JSONDecodeError, OSError):
                self._memory = {}

    def _save(self) -> None:
        cf = self._cache_file()
        cf.write_text(json.dumps(self._memory), encoding="utf-8")

    @staticmethod
    def _hash(content: str) -> str:
        return hashlib.sha256(content.encode("utf-8")).hexdigest()

    def get(self, path: str, content: str) -> Optional[list]:
        key = f"{path}:{self._hash(content)}"
        entry = self._memory.get(key)
        if entry is not None:
            return entry
        return None

    def put(self, path: str, content: str, findings: list) -> None:
        key = f"{path}:{self._hash(content)}"
        self._memory[key] = findings

    def flush(self) -> None:
        self._save()
```

```python
#!/usr/bin/env python3
"""Benchmark Regula against real OSS AI projects.

Clones 5 projects, runs regula check, outputs results for labelling.
"""
import json
import subprocess
import sys
import tempfile
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent.parent / "scripts"))

PROJECTS = [
    {"name": "instructor", "repo": "https://github.com/jxnl/instructor.git", "shallow": True},
    {"name": "pydantic-ai", "repo": "https://github.com/pydantic/pydantic-ai.git", "shallow": True},
    {"name": "langchain", "repo": "https://github.com/langchain-ai/langchain.git", "shallow": True},
    {"name": "scikit-learn", "repo": "https://github.com/scikit-learn/scikit-learn.git", "shallow": True},
    {"name": "openai-python", "repo": "https://github.com/openai/openai-python.git", "shallow": True},
]


def clone_project(project: dict, base_dir: Path) -> Path:
    dest = base_dir / project["name"]
    if dest.exists():
        return dest
    cmd = ["git", "clone", "--depth=1", project["repo"], str(dest)]
    subprocess.run(cmd, capture_output=True, timeout=120)
    return dest


def scan_project(project_path: str) -> list:
    from report import scan_files
    return scan_files(project_path)


def main():
    output_dir = Path(__file__).parent / "results"
    output_dir.mkdir(exist_ok=True)

    with tempfile.TemporaryDirectory() as tmp:
        tmp_path = Path(tmp)
        for project in PROJECTS:
            print(f"\n{'='*60}")
            print(f"Benchmarking: {project['name']}")
            print(f"{'='*60}")

            project_path = clone_project(project, tmp_path)
            findings = scan_project(str(project_path))

            active = [f for f in findings if not f.get("suppressed")]
            print(f"  Total findings: {len(active)}")

            # Categorise
            by_tier = {}
            for f in active:
                tier = f.get("tier", "unknown")
                by_tier.setdefault(tier, []).append(f)

            for tier, items in sorted(by_tier.items()):
                print(f"  {tier}: {len(items)}")

            # Save for labelling
            output_file = output_dir / f"{project['name']}.json"
            output_file.write_text(json.dumps(active, indent=2, default=str), encoding="utf-8")
            print(f"  Saved to: {output_file}")

    print(f"\nBenchmark complete. Results in {output_dir}/")
    print("Next: manually label each finding as TP/FP/DISPUTED in the JSON files.")


if __name__ == "__main__":
    main()
```
