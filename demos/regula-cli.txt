import openai
def classify_resume(resume_text):
    client = openai.OpenAI()
    reply = client.chat.completions.create(
        model='gpt-4',
        messages=[{'role':'user','content':f'Score this resume: {resume_text}'}])
    return reply.choices[0].message.content
